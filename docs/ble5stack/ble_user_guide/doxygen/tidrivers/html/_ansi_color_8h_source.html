69 /* @brief Returns an ANSI escape sequence for the chosen colors and style
70  *
71  * The parameters are variadic, and up to five can be specified in any order:
72  *
73  * FG_BLACK, FG_RED, FG_GREEN, FG_YELLOW, FG_BLUE, FG_MAGENTA, FG_CYAN, FG_WHITE,
74  * BG_BLACK, BG_RED, BG_GREEN, BG_YELLOW, BG_BLUE, BG_MAGENTA, BG_CYAN, BG_WHITE,
75  * ATTR_RESET, ATTR_BOLD, ATTR_UNDERLINE, ATTR_BLINK
76  *
77  * Example use:
78  * char str[] = ANSI_COLOR(FG_WHITE, ATTR_BOLD) "Hello" ANSI_COLOR(ATTR_RESET)
79  * --> "Esc[37;1mHelloEsc[0m" where Esc is the Escape control character 0x1B.
80  *
81  * @param 1 - color or style
82  * @param 2 - color or style
83  * @param 3 - color or style
84  * @param 4 - color or style
85  * @param 5 - color or style
86  *
87  */

92  _ANSI_TERM_CONCAT_COLOR3, \
