79 /*
80  * Performance of the ECC SW library is scaled via the window size parameter
81  * which defines the number of bits of the scalar that will be processed together
82  * during the execution of the scalar multiplication. A larger window size
83  * will have higher performance at the cost of increased memory consumption.
84  * A window size of 3 was selected for the best trade-off of performance and
85  * memory consumption.
86  *
87  * ---------------------------------------------------
88  * | | NIST P256 | X25519 |
89  * | Window Size | WorkZone Size | WorkZone Size |
90  * | | (words) | (words) |
91  * |-------------|-----------------|-----------------|
92  * | 2 or 3 | 171 | 198 |
93  * | 4 | 225 | 225 |
94  * | 5 | 333 | 333 |
95  * ---------------------------------------------------
96  */
97 #define ECDH26X1_ECC_WINDOW_SIZE 3
98 
99 /*
100  * ECC Workzone size is based on worst case empirical measurement of ECDH
101  * operations on supported curves with the selected window size.
102  * NIST P256 (win_size=3): 171 words
103  * Curve25519 (win_size=3): 198 words
104  */
105 #define ECDH26X1_ECC_WORKZONE_WORDS 198
