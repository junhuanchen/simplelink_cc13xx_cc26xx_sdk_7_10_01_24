352  class DiscoveredPrefixTable : public InstanceLocator
353  {
354  // This class maintains the discovered on-link and route prefixes
355  // from the received RA messages by processing PIO and RIO options
356  // from the message. It takes care of processing the RA message but
357  // delegates the decision whether to include or exclude a prefix to
358  // `RoutingManager` by calling its `ShouldProcessPrefixInfoOption()`
359  // and `ShouldProcessRouteInfoOption()` methods.
360  //
361  // It manages the lifetime of the discovered entries and publishes
362  // and unpublishes the prefixes in the Network Data (as external
363  // route) as they are added or removed.
364  //
365  // When there is any change in the table (an entry is added, removed,
366  // or modified), it signals the change to `RoutingManager` by calling
367  // `HandleDiscoveredPrefixTableChanged()` callback. A `Tasklet` is
368  // used for signalling which ensures that if there are multiple
369  // changes within the same flow of execution, the callback is
370  // invoked after all the changes are processed.
371 
372  public:
373  enum NetDataMode : uint8_t // Used in `Remove{}` methods
374  {
375  kUnpublishFromNetData, // Unpublish the entry from Network Data if previously published.
376  kKeepInNetData, // Keep entry in Network Data if previously published.
377  };
